/**
 * @fileoverview Wrapper for automatic inclusion of all Jala modules.
 */


// Define the global namespace for Jala modules
if (!global.jala) {
   global.jala = {};
}


(function() {
   var packages = [
      "AsyncRequest",
      "BitTorrent",
      "Date",
      "DnsClient",
      "Captcha", 
      "History",
      "HtmlDocument",
      "HopObject",
      "i18n",
      "ImageFilter",
      "IndexManager",
      "ListRenderer",
      "Mp3Info",
      "PodCastWriter",
      "RemoteContent",
      "Rss20Writer",
      "Utilities",
      "XmlRpcRequest",
      "XmlWriter"
   ];
   var jalaDir = getProperty("jala.dir", "modules/jala/code/");
   for (var i in packages) {
      app.addRepository(jalaDir + packages[i] + ".js");
   }
   return;
})();


/**
 * Get a string representation of the Jala library.
 * @returns [Jala JavaScript Application Library]
 * @type String
 */
jala.toString = function() {
   return "[Jala JavaScript Application Library]";
};
